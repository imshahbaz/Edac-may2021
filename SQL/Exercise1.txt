create table Supplier(
Sno varchar(5),
Sname varchar(20),
Status varchar(2),
City varchar(20),
primary key(Sno)
);

create table Parts(
Pno varchar(5),
Pname varchar(20),
Color varchar(10),
Weight float(5,2),
City varchar(20),
primary key(Pno)
);

create table Projects(
Jno varchar(5),
Jname varchar(20),
City varchar(20),
primary key(Jno)
);

create table SPJ(
Sno varchar(5),
Pno varchar(5),
